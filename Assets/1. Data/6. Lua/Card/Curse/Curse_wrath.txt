Curse_wrath = class(CardBase,'Curse_wrath')
local this = Curse_wrath

function this:new()
  self.cardView = nil
  self.name = "诅咒-愤怒"
  self.frame = "诅咒牌面"
  self.image = "残损义骸"
  self.desc = "无效果，无法打出，无能狂怒"
  --self.usageAmount = 1
  self.inHand = false
end


function this:onPlay()
  self.cardView:Dissolve()
  --Debug.Log(GameData.drawPile)
end
