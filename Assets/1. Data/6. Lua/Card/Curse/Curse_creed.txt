Curse_creed = class(CardBase,'Curse_creed')
local this = Curse_creed

function this:new()
  self.cardView = nil
  self.name = "诅咒-贪婪"
  self.frame = "诅咒牌面"
  self.image = "贪欲"
  self.desc = "随机获得下列效果中的一种：获得2能量/失去4能量/抽1张卡/卡组中加入1张贪婪"
  --self.usageAmount = 1
  self.inHand = false
end


function this:onPlay()
  GameManager.Discard(self)
  local i = math.random(1,4)
  if i==1 then GameManager.AddResource(ResourceType.Mine,2)
  elseif i==2 then GameManager.SubResource(ResourceType.Mine,4)
  elseif i==3 then GameManager.DrawDeck()
  else GameManager.AddCard(DeckType.OriginalDeckAndDrawPile,{"Curse_creed"})
  end
end
