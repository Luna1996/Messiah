Curse_creed = class(CardBase,'Curse_creed')
local this = Curse_creed

function this:new()
  self.cardView = nil
  self.name = "诅咒-贪婪"
  self.frame = "诅咒牌面"
  self.image = "贪欲"
  self.desc = "获得下列效果之一：\n获得8点能量\n失去4点能量\n抽1张牌\n牌组中加入1张<b>[贪婪]</b>"
  --self.usageAmount = 1
  self.inHand = false
end


function this:onPlay()
  --效果待写
  GameManager.Discard(self)
  local r = math.random(1,4)
  if r==1 then GameManager.AddResource(ResourceType.Mine,8)
  elseif r==2 then GameManager.SubResource(ResourceType.Mine,4)
  elseif r==3 then GameManager.DrawCard()
  else GameManager.AddCard(DeckType.OriginalDeckAndDrawPile,{"Curse_creed",}) end
end
