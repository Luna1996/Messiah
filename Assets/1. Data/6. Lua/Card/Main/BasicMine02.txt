local this = BasicMine02

function this:new()

  self.name = "采矿2"
  --self.frame = "CardFrame_0"

  self.e1 = ResourceModifier("Mine",2)

  --LuaGameData.Resource.Mine = LuaGameData.Resource.Mine +2
  self.e1.apply()

end
