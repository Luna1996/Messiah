BasicMine01 = class(CardBase,'BasicMine01')
local this = BasicMine01

function this:new()
  self.cardView = nil
  self.name = "采矿1"
  --self.frame = "CardFrame_0"
  self.image = "采矿1"
  self.desc = "消耗1点人力 获得1点矿石"

  self.inHand = false
end

function this:onPlay()
  --self.usageAmount = self.usageAmount - 1
  GameManager.AddResource(ResourceType.Mine, 1)
  GameManager.Discard(self)
end
