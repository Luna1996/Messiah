BasicMine01 = class(CardBase,'BasicMine01')
local this = BasicMine01

function this:new()
  self.cardView = nil
  self.name = "采矿1"
  --self.frame = "CardFrame_0"
  self.image = "采矿1"
  self.desc = "消耗1点人力 获得1点矿石"

  self.inHand = false
  self.e1 = ResourceModifier("Mine",2)
  self.cost = LaborConsumer
end


function this:onPlay()
  self.cardView:Dissolve()
  --Debug.Log(GameData.drawPile.Count)
  --Debug.Log(LuaGameData.Resource.Mine)
  --LuaGameData.Resource.Mine = LuaGameData.Resource.Mine +2
  self.e1:apply()
  Debug.Log('CurrentMine:'..LuaGameData.Resource.Mine)
end
