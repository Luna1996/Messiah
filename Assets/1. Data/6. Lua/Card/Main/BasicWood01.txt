BasicWood01 = class(CardBase,'BasicWood01')
local this = BasicWood01

function this:new()
  self.cardView = nil
  self.name = "采石1"
  --self.frame = "CardFrame_0"
  self.frame = "资源牌面s"
  self.image = "采石1"
  self.desc = "下回合,获得5点石材"
  self.cost = 1
  self.rounds = 1
  self.inHand = false
end

function this:applyEffect()
  EventService.Notify(GameEvent.IG_ResourceCardPlay)
  Buff(
    "Buff_BasicWood01", 
    GameEvent.EnterMainPhase, 
    BuffType.OneShot, 
    1, "buff503", nil,
    "Xturn", "1,获得5点石材")
  GameManager.Discard(self)
end
