BasicIron01 = class(CardBase,'BasicIron01')
local this = BasicIron01
--Test：人口显示测试
function this:new()
  self.cardView = nil
  self.name = "炼钢1"
  --self.frame = "CardFrame_0"
  self.image = "炼钢1"
  self.desc = "消耗1点人力，获得1点钢铁"

  self.inHand = false
  --self.e1 = ResourceModifier("Iron",2)
  --self.cost = LaborConsumer
end

function this:onPlay()
  if (GameManager.gameData.idleWorker >= 1)
  then
    GameManager.gameData.idleWorker = GameManager.gameData.idleWorker - 1
    GameManager.AddResource(ResourceType.Iron, 1)
    GameManager.CreateBuff(GameEvent.EnterMainPhase, BuffType.OneShot, 1, 
    function()
      GameManager.gameData.idleWorker = GameManager.gameData.idleWorker + 1
    end)
    GameManager.Discard(self)
  else
    HandView:AddToHand(self.cardView)
  end
end
