BasicMush01 = class(CardBase,'BasicMush01')
local this = BasicMush01

function this:new()
  self.cardView = nil
  self.name = "蘑菇培植1"
  --self.frame = "CardFrame_0"
  self.image = "蘑菇培植1"
  self.desc = "消耗1点人力2回合 获得10点食物"

  self.inHand = false
  self.e1 = ResourceModifier("Iron",2)
  self.cost = LaborConsumer
end


function this:onPlay()
  self.cardView:Dissolve()
  --Debug.Log(GameData.drawPile.Count)
  --Debug.Log(LuaGameData.Resource.Mine)
  --LuaGameData.Resource.Mine = LuaGameData.Resource.Mine +2
  self.e1.apply()
  Debug.Log(LuaGameData.Resource.Mine)
end
