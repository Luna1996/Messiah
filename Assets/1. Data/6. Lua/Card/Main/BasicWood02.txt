BasicWood02 = class(CardBase,'BasicWood02')
local this = BasicWood02

function this:new()
  self.cardView = nil
  self.name = "采石2"
  --self.frame = "CardFrame_0"
  self.frame = "资源牌面s"
  self.image = "采石2"
  self.desc = "下回合,获得8点石材"
  self.cost = 2
  self.rounds = 1
  self.inHand = false
end


function this:applyEffect()
  Buff(
    "Buff_BasicWood02", 
    GameEvent.EnterMainPhase, 
    BuffType.OneShot, 
    1, "buff503", nil,
    "Xturn", "1,获得8点石材")
  GameManager.Discard(self)
end
