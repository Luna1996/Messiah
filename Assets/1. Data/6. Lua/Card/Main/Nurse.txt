Nurse = class(CardBase,'Nurse')
local this = Nurse


function this:new()
  self.cardView = nil
  self.name = "看护"
  self.frame = "CardFrame_1"
  self.image = "看护1"
  self.desc = "借用神明之力，救回迷雾侵染之人\n消耗食物3 能量3 信仰3\n成功概率：30%"

  self.foodc=3

end

function this:applyEffect()

end
