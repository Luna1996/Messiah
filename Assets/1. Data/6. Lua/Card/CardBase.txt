  self.frame = "CardFrame_1"

  self.desc = "<color=red>规</color><b><color=green>则</color><color=blue>说</color></b>明"

  self.inHand = false
end

end

end

end
    
function this:rmvFrHand()
end

function this:getCardFullName()
  return self.className
end

function this:canBuild(buildingConsumption)
  local woodc = buildingConsumption[1]
  local ironc = buildingConsumption[2]
  local workerc = buildingConsumption[3]
  if woodc<=GameManager.GetResource(ResourceType.Wood) and ironc<=GameManager.GetResource(ResourceType.Iron) and workerc<=GameManager.gameData.idleWorker then
    GameManager.SubResource(ResourceType.Wood,woodc)
    GameManager.SubResource(ResourceType.Iron,ironc)
    GameManager.gameData.idleWorker = GameManager.gameData.idleWorker - workerc
    return true
  else
    return false
  end
end
