Nurse = class(CardBase,'Nurse')
local this = Nurse


function this:new()
  self.cardView = nil
  self.name = "看护"
  --self.frame = "CardFrame_0"
  self.image = "看护1"
  self.desc = "抽2张牌"
  self.temp = true
  self.inHand = false
end

function this:onPlay()
  GameManager.Discard(self)
  GameManager.DrawCard()
  GameManager.DrawCard()
end
