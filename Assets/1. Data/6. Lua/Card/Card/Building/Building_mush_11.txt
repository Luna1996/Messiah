Building_mush_11= class(CardBase,"Building_mush_11")
local this = Building_mush_11

function this:new()
  self.cardView = nil
  self.name = "孢子克隆实验室"
  self.frame = "资源牌面s"
  self.image = "蘑菇pd"
  self.desc = "消耗石材6，钢铁2\n可以培植巨型真菌"
  self.inHand = false
  self.woodc=6
  self.ironc=2
  self.cost=2
  self.rounds=3
end


function this:applyEffect()
  self.cardView:Dissolve()
  Buff("Build_mush_11",
  GameEvent.EnterMainPhase,
  BuffType.OneShot,
  self.rounds,
  'buff518',
  nil,
  "BuildTip",
  "3,孢子克隆实验中")
  --Debug.Log(GameData.drawPile)
end
