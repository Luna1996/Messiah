Tech_build_2= class(CardBase,"Tech_build_2")
local this = Tech_build_2

function this:new()
  self.cardView = nil
  self.name = "高级建造"
  self.frame = "科技牌面"
  self.image = "高级建造科技"
  self.desc = "消耗12科技点数 将【天气控制装置】【温室】【温室升级】加入建筑牌堆"  ---科技卡分两种：一种将其他科技卡加入手牌抽牌堆，一种将科技建筑卡加入建筑牌堆
  self.boundingCard={"Building_weather_21","Building_farm","Building_farm_21"}
  self.inHand = false
  --self.boundingBuff={}
  --self.buildingConsumption={2,0,2,2}
end


function this:onPlay()
  if GameManager.GetResource(ResourceType.Tech)>=12 then
    GameManager.SubResource(ResourceType.Tech,12)
    --Debug.Log(GameData.buildingDeck.Count)
    GameManager.Exile(self)
    GameManager.AddCard(DeckType.BuildingDeck,self.boundingCard)
    --Debug.Log(GameData.buildingDeck.Count)
  else
    HandView:AddToHand(self.cardView)
  end

end
