Building_mine_21 = class(CardBase,"Building_mine_21")
local this = Building_mine_21

function this:new()
  self.cardView = nil
  self.name = "连轴运转"
  self.frame = "建筑牌面"
  self.image = "残损义骸"
  self.desc = "所有资源卡获得装填" --装填：打出时再抽一张牌
  self.inHand = false
  self.buildingConsumption={8,10,6,3}
  self.cost =self.buildingConsumption[3]
  self.rounds =self.buildingConsumption[4]
end

---buff trigger加一个打出资源牌时drawcard

function this:onPlay()
  self.cardView:Dissolve()
  --Debug.Log(GameData.drawPile)
end
