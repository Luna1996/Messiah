Building_iron = class(CardBase,"Building_iron")
local this = Building_iron

function this:new()
  self.cardView = nil
  self.name = "钢铁厂"
  self.frame = "资源牌面s"
  self.image = "钢厂p"
  self.desc = "消耗石材3,钢铁3\n 获取钢材的生产场所\n牌组中加入2张炼钢牌"
  self.boundingCard={"BasicIron01","BasicIron01"}
  self.inHand = false
  --self.boundingBuff={}
  self.woodc=8
  self.ironc=4
  self.cost =7
  self.rounds =3
end

function this:applyEffect()
  self.cardView:Dissolve()
  Buff("Build_iron_basic",GameEvent.EnterMainPhase,BuffType.OneShot,3,'buff518',nil,"BuildTip","3,正在建造钢厂")
end
