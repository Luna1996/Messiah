Building_house= class(CardBase,"Building_house")
local this = Building_house

function this:new()
  self.cardView = nil
  self.name = "住宅"
  self.frame = "资源牌面s"
  self.image = "住宅p"
  self.desc = "-石材2 钢铁0- 普通的民宅，城市的人口数量+1"
  --self.boundingCard={"Pray01","Pray02","Pray02"}
  self.inHand = false
  self.boundingBuff={}
  self.buildingConsumption={2,0,2,1}
  self.cost =self.buildingConsumption[3]
  self.rounds =self.buildingConsumption[4]
end

function this:onPlay()
  if self:enoughLabor(self.cost+CostModifiter) and self:enoughRcBuilding(self.buildingConsumption) then
    self:subRcBuilding(self.buildingConsumption)
    self:laborCost(self.cost+CostModifiter,self.rounds)
      --get一下当前空闲人力
      GameManager.gameData.MaxWorker = GameManager.gameData.MaxWorker + 1
      GameManager.Exile(self)
  else
    HandView:AddToHand(self.cardView)
  end
end
