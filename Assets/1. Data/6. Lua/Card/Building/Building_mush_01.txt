Building_mush_01= class(CardBase,"Building_mush_01")
local this = Building_mush_01

function this:new()
  self.cardView = nil
  self.name = "蘑菇水培池"
  self.frame = "建筑牌面"
  --self.image = "残损义骸"
  self.desc = "-石材5 钢铁2- 提高蘑菇农场的蘑菇卡等级"
  self.inHand = false
  self.boundingCard="BasicMush02"
  self.deletingCard="BasicMush01"
  self.cost =2
  self.rounds =2
  self.woodc=5
  self.ironc=2

end

function this:applyEffect()
  self.cardView:Dissolve()
  Buff("Build_mush_01",
  GameEvent.EnterMainPhase,
  BuffType.OneShot,
  self.rounds,
  'buff518',
  nil,
  "BuildTip",
  "2,蘑菇水培池建造中")
end
