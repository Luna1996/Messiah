Building_mine= class(CardBase,"Building_mine")
local this = Building_mine

function this:new()
  self.cardView = nil
  self.name = "矿场"

  self.frame = "资源牌面s"
  self.image = "采煤机p"
  self.desc = "消耗石材3，钢铁2\n开采矿石的场所\n卡组中加入2张采能量卡"

  self.boundingCard={"BasicMine01","BasicMine01"}
  self.inHand = false
  --self.boundingBuff={}

  self.woodc=3
  self.ironc=0
  self.cost=3
  self.rounds=2
end

function this:applyEffect()
  self.cardView:Dissolve()
  Buff("Build_mine_basic",
  GameEvent.EnterMainPhase,
  BuffType.OneShot,
  2,
  'buff518',
  nil,
  "2,采集能源矿机建设中")
end
