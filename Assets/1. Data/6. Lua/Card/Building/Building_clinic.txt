Building_clinic = class(CardBase,"Building_clinic")
local this = Building_clinic

function this:new()
  self.cardView = nil
  self.name = "诊所"
  self.frame = "建筑牌面"
  self.image = "残损义骸"
  self.desc = "卡组中加入1张看护"
  self.boundingCard={"Nurse",}
  self.inHand = false
  self.boudingBuff={}
  self.buildingConsumption={8,3,5,2}
end

function this:onPlay()
  GameManager.Exile(self)
  for i,v in ipairs(self.boundingCard) do
    GameData.drawPile.Add(v)
  end
end
