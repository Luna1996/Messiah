Building_clinic = class(CardBase,"Building_clinic")
local this = Building_clinic

function this:new()
  self.cardView = nil
  self.name = "诊所"
  self.frame = "资源牌面s"
  self.image = "医院p"
  self.desc = "消耗石材8,钢铁3\n卡组中加入1张【看护】"
  self.boundingCard={"Nurse",}
  self.inHand = false
  self.boudingBuff={}
  self.woodc = 8
  self.ironc = 3
  self.cost =5
  self.rounds =2
end

function this:applyEffect()
    GameManager.Exile(self)
    GameManager.AddCard(DeckType.OriginalDeckAndDrawPile,self.boundingCard)
end
