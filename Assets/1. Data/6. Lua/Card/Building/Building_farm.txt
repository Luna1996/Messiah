Building_farm= class(CardBase,"Building_farm")
local this = Building_farm

function this:new()
  self.cardView = nil
  self.name = "温室"
  self.frame = "建筑牌面"
  self.image = "残损义骸"
  self.desc = "更高质量的食物来源"
  self.boundingCard={"BasicFarm01","BasicFarm01"}
  self.inHand = false
  --self.boundingBuff={}
  self.buildingConsumption={20,20,10,4}
end

function this:onPlay()
  GameManager.Exile(self)
  for i,v in ipairs(self.boundingCard) do
    GameData.drawPile.Add(v)
  end
end
