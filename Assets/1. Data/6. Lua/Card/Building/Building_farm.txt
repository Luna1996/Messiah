Building_farm= class(CardBase,"Building_farm")
local this = Building_farm

function this:new()
  self.cardView = nil
  self.name = "温室"
  self.frame = "资源牌面s"
  self.image = "温室p"
  self.desc = "消耗钢铁20,石材20\n获得更高质量的食物来源\n卡组中加入2张【农作物种植1】"
  self.boundingCard={"BasicFarm01","BasicFarm01"}
  self.inHand = false
  self.buildingConsumption={20,20,10,4}
  self.cost =self.buildingConsumption[3]
  self.rounds =self.buildingConsumption[4]
end

function this:applyEffect()
  GameManager.Exile(self)
  GameManager.AddCard(DeckType.OriginalDeckAndDrawPile,self.boundingCard)
end
