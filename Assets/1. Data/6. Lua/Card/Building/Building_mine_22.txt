Building_mine_22 = class(CardBase,"Building_mine_22")
local this = Building_mine_22

function this:new()
  self.cardView = nil
  self.name = "源能切割"
  self.frame = "建筑牌面"
  self.image = "残损义骸"
  self.desc = "每打出1张资源卡额外消耗1点矿石，所有资源卡产量+2"
  self.inHand = false
  self.boundingBuff={} --即时生效 为牌组内所有资源卡增加上述效果
  self.buildingConsumption={12,10,5,3}
  self.woodc=12
  self.ironc=10
  self.cost=5
  self.rounds=3
end

function this:onPlay()
  self.cardView:Dissolve()
  Buff("Build_mine_22",
  GameEvent.EnterMainPhase,
  BuffType.OneShot,
  3,
  'buff518',
  nil,
  "BuildTip",
  "3,源能切割研究中")
end
