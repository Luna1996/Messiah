Building_church_01 = class(CardBase,"Building_church_01")
local this = Building_church_01

function this:new()
  self.cardView = nil
  self.name = "教堂修缮"
  self.frame = "资源牌面s"
  self.image = "残损义骸"
  self.desc = "-石材10 钢铁0- 进一步修缮教堂，为大家提供祷告的场所（牌组中加入1张【祈祷】、2张【祈愿】）"
  self.boundingCard={"Pray01","Pray02","Pray02"}
  self.inHand = false
  self.woodc = 10
  self.cost =3
  self.rounds =2
end
--重写this:applyEffect()
function this:applyEffect()
    GameManager.Exile(self)
    GameManager.AddCard(DeckType.OriginalDeckAndDrawPile,self.boundingCard)
end
