-- EventExample


local states ={}

states.a = {
  name = "求救事件D1",
  desc = "有人呼救",
  choices = {"花费随机2点资源前往救援","置之不理"},
}


function states.a:choose(choice)
    if choice == self.choices[1] then
      --随机挑选1种资源进行消耗
      --if 不够则 rerturn nil，else随机获得2点资源
      return nil
    elseif choice==self.choices[2] then
      --调用弹对话的接口 text= "狠心！！！"
      return nil
    end
end


Event1051 = Event(states.a)
