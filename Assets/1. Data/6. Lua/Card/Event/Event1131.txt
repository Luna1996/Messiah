-- EventExample


local states ={}

states.a = {
  name = "旅行者B1",
  desc = "疲惫的旅行者 需要xx资源",
  choices = {"给他","8给"},
}

function states.a:choose(choice)
    if choice == self.choices[1] then
      --卡牌奖励池1
      --随机选择资源2-5点进行扣减 不足的话直接nil
      --足够扣减 则获得随机数量的信仰值
      return nil
    elseif choice==self.choices[2] then
      --调用弹对话的接口 无事发生
      return nil
    end
end

Event1111 = Event(states.a)
