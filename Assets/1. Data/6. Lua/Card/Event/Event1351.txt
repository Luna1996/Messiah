-- EventE1351


local states ={}

states.a = {
  name = "约翰退休",
  desc = "我要退休",
  choices = {"好的","不行"},
  }
function states.a:choose(choice)
    if choice == self.choices[1] then
      --调用人口-1，同时不满值降低5点
      return nil
    elseif choice==self.choices[2] then
      --调用弹对话的接口 text= "你坏坏！"
      --不满值上升5点
      return nil
    end
end

EventExample = Event(states.a)
