  name = "疯婆",
  desc = "需要x点资源",

  }

    if choice == self.choices[1] then
      --判断资源是否够用
        --获得一个强力的buff
      return nil
    elseif choice==self.choices[2] then
      --信仰值降低
      --调用弹对话的接口 text= "你坏坏！"
      return nil
    end
end
