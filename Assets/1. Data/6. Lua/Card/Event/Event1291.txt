local states ={}

states.a = {
  name = "万圣节的请求",
  desc = "消耗10点食物，制作南瓜灯",
  choices = {"嗯呢","不给"},
  }
function states.a:choose(choice)
    if choice == self.choices[1] then
      --判断食物资源是否够用
        --受到万圣节的气氛，每回合增加2点食物
      return nil
    elseif choice==self.choices[2] then
      --调用弹对话的接口 text= "你坏坏！"
      return nil
    end
end

EventExample = Event(states.a)
