Relic311 = Relic()
local this = Relic311
this.ID=311
this.name='深渊赠礼'
this.desc='随机获得3张牌进入卡组'
this.rare='pandora'
this.icon=nil

function this.apply()
  local randomList={"Curse_lust","BasicFoodPile","BasicIronPile","BasicMinePile","BasicWoodPile","Building_house",
  "Building_mine","Building_wood","Curse_creed","Curse_gluttony","BasicCorePile"}
  for i=1,3 do
    GameManager.AddCard(DeckType.DrawPile,{randomList[math.random(1,#randomList)]})
  end
end
