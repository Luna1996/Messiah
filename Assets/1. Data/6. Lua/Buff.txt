function Xturn(i, args)
  local t = split(args,',')
  return "下"..t[1]-i.."回合,"..t[2], t[1]-i
end

function XLaborTurn(i, args)
  local t = split(args,',')
  return t[1]..t[2], t[3]-i
end

function AddBackLabor(i, args)
  local t = split(args, ',')
  GameManager.gameData.idleWorker = GameManager.gameData.idleWorker + tonumber(t[1])
end
